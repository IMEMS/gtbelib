77 void synth_init(bool synthEN, uint8_t xtal_cl, uint8_t* regConfig, uint8_t outputEn) {
78  synth_setXTALcapacitance(xtal_cl);
79  if(synthEN) {
80  synth_initADCreclkENL(false);
81  synth_initConfig(regConfig, outputEn);

103 void synth_initConfig(uint8_t* regConfig, uint8_t outputEn) {
104  // Disable Outputs
105  synth_outputDisableAll();
106  // Powerdown all output drivers
107  synth_powerDownOutputDrivers();
108  // Set interrupt masks (Mask all interrupts)
109  synth_setInterruptMasks(SYNTH_INT_STATUS_MASK_SYS_INIT_MASK |
110  SYNTH_INT_STATUS_MASK_LOL_B_MASK |
111  SYNTH_INT_STATUS_MASK_LOL_A_MASK |
112  SYNTH_INT_STATUS_MASK_LOS_MASK);
113  synth_writeRegConfig(regConfig);
114  // Apply PLLA and PLLB soft reset
115  synth_PLLreset(SYNTH_RST_PLL_PLLB_RST | SYNTH_RST_PLL_PLLA_RST);
116  // Enable desired outputs (enables CLK0)
117  synth_outputEnable(outputEn);
118 }

293 void synth_initI2C(bool fast) {
294  // Enable Peripherals
295  MAP_SysCtlPeripheralEnable(SYNTH_I2C_PERIPH);
296  MAP_SysCtlPeripheralEnable(SYNTH_GPIO_PERIPH);
297  // Set the pin muxing for I2C peripheral
298  MAP_GPIOPinConfigure(SYNTH_I2C_SCL_PIN_CONFIG);
299  MAP_GPIOPinConfigure(SYNTH_I2C_SDA_PIN_CONFIG);
300 
301  MAP_GPIOPinTypeI2CSCL(SYNTH_GPIO_BASE, SYNTH_SCL_PIN);
302  MAP_GPIOPinTypeI2C(SYNTH_GPIO_BASE, SYNTH_SDA_PIN);
303 
304  I2CMasterInitExpClk(SYNTH_I2C_BASE, SysCtlClockGet(), fast); // true = 400kbps, false = 100kbps
305  I2CMasterSlaveAddrSet(SYNTH_I2C_BASE, SYNTH_I2C_SLAVE_ADDRESS, false);
306 }

314 void synth_writeRegI2C(uint8_t regNum, uint8_t regValue) {
315  // Write register number
316  while(I2CMasterBusy(SYNTH_I2C_BASE)) {} // Wait until the I2C line is not busy
317  I2CMasterSlaveAddrSet(SYNTH_I2C_BASE, SYNTH_I2C_SLAVE_ADDRESS, false);
318  I2CMasterDataPut(SYNTH_I2C_BASE, regNum);
319  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_BURST_SEND_START);
320  twe_I2CMasterVerify(SYNTH_I2C_BASE, true ,false);
321  // Write register value
322  I2CMasterDataPut(SYNTH_I2C_BASE, regValue);
323  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_BURST_SEND_FINISH);
324  twe_I2CMasterVerify(SYNTH_I2C_BASE, true ,false);
325 }

335 void synth_writeConsecutiveRegsI2C(uint8_t regNumStart, uint8_t regCNT, uint8_t* regValues) {
336  ASSERT(regCNT > 1);
337 
338  // Write first register number
339  while(I2CMasterBusy(SYNTH_I2C_BASE)) {} // Wait until the I2C line is not busy
340  I2CMasterSlaveAddrSet(SYNTH_I2C_BASE, SYNTH_I2C_SLAVE_ADDRESS, false);
341  I2CMasterDataPut(SYNTH_I2C_BASE, regNumStart);
342  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_BURST_SEND_START);
343  twe_I2CMasterVerify(SYNTH_I2C_BASE, true ,false);
344 
345  // Write each register except the last
346  uint32_t i = 0;
347  while(i < regCNT-1) {
348  I2CMasterDataPut(SYNTH_I2C_BASE, regValues[i]);
349  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_BURST_SEND_CONT);
350  twe_I2CMasterVerify(SYNTH_I2C_BASE, true ,false);
351  i++;
352  }
353 
354  // Write last register
355  I2CMasterDataPut(SYNTH_I2C_BASE, regValues[i]);
356  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_BURST_SEND_FINISH);
357  twe_I2CMasterVerify(SYNTH_I2C_BASE, true ,false);
358 }

367 uint32_t synth_readRegI2C(uint8_t regNum) {
368  while(I2CMasterBusy(SYNTH_I2C_BASE)) {} // Wait until the I2C line is not busy
369  // Write register number
370  I2CMasterSlaveAddrSet(SYNTH_I2C_BASE, SYNTH_I2C_SLAVE_ADDRESS, false);
371  I2CMasterDataPut(SYNTH_I2C_BASE, regNum);
372  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_SINGLE_SEND);
373  twe_I2CMasterVerify(SYNTH_I2C_BASE, false ,false);
374  // Read register value
375  I2CMasterSlaveAddrSet(SYNTH_I2C_BASE, SYNTH_I2C_SLAVE_ADDRESS, true);
376  I2CMasterControl(SYNTH_I2C_BASE, I2C_MASTER_CMD_SINGLE_RECEIVE);
377  twe_I2CMasterVerify(SYNTH_I2C_BASE, false ,true);
378  return I2CMasterDataGet(SYNTH_I2C_BASE);
379 }
