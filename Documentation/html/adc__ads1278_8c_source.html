186  void ADC_setCLKfreq(uint32_t clkFreqValue) {
187  switch(clkFreqValue) {
188  case ADC_FREQ_37MHZ_HS :
189  ADC_setMode(ADC_MODE_HIGH_SPEED);
190  ADC_setCLKdiv(ADC_CLKDIV_HIGHF);
191  break;
192  case ADC_FREQ_27MHZ_HR :
193  ADC_setMode(ADC_MODE_HIGH_RESOL);
194  ADC_setCLKdiv(ADC_CLKDIV_HIGHF);
195  break;
196  case ADC_FREQ_27MHZ_LP :
197  ADC_setMode(ADC_MODE_LOW_POWER);
198  ADC_setCLKdiv(ADC_CLKDIV_HIGHF);
199  break;
200  case ADC_FREQ_13p5MHZ_LP :
201  ADC_setMode(ADC_MODE_LOW_POWER);
202  ADC_setCLKdiv(ADC_CLKDIV_LOWF);
203  break;
204  case ADC_FREQ_27MHZ_LS :
205  ADC_setMode(ADC_MODE_LOW_SPEED);
206  ADC_setCLKdiv(ADC_CLKDIV_HIGHF);
207  break;
208  case ADC_FREQ_5p4MHZ_LS :
209  ADC_setMode(ADC_MODE_LOW_SPEED);
210  ADC_setCLKdiv(ADC_CLKDIV_LOWF);
211  break;
212  default :
213  ADC_setMode(ADC_MODE_HIGH_SPEED);
214  ADC_setCLKdiv(ADC_CLKDIV_HIGHF);
215  }
216  }

258  void ADC_initClkPWM(uint32_t sysClkFreqDiv) {
259  // Setup GPIO config
260  MAP_SysCtlPeripheralEnable(ADC_PWM_PERIPH);
261  MAP_SysCtlPeripheralEnable(ADC_PWM_GPIO_PERIPH);
262 
263  MAP_GPIOPinTypePWM(ADC_PWM_GPIO_BASE, ADC_PWM_PIN);
264  MAP_GPIOPinConfigure(ADC_PWM_PIN_CONFIG);
265  SysCtlPWMClockSet(SYSCTL_PWMDIV_1);
266 
267  // Configure the PWM generator for count down mode with immediate updates
268  // to the parameters.
269  MAP_PWMGenConfigure(ADC_PWM_BASE, ADC_PWM_GEN,
270  PWM_GEN_MODE_DOWN | PWM_GEN_MODE_NO_SYNC | PWM_GEN_MODE_DBG_STOP);
271 
272  #ifdef PART_TM4C123GH6PM // EK-TM4C123GXL
273  // Set the period in number of clock cycles.
274  MAP_PWMGenPeriodSet(ADC_PWM_BASE, ADC_PWM_GEN, 4); //works freq = 20MHz
275  //PWMGenPeriodSet(PWM1_BASE, PWM_GEN_2, 8); //works freq = 10MHz
276  // Set the pulse width.
277  MAP_PWMPulseWidthSet(ADC_PWM_BASE, ADC_PWM_OUT, 2); //works freq = 20MHz
278  //PWMPulseWidthSet(PWM1_BASE, PWM_OUT_5, 4); //works freq = 10MHz
279  #endif
280  #ifdef PART_TM4C1294NCPDT // EK-TM4C1294XL
281  // Set the period in number of clock cycles.
282  MAP_PWMGenPeriodSet(ADC_PWM_BASE, ADC_PWM_GEN, 6); //works freq = 20MHz
283  // Set the pulse width.
284  MAP_PWMPulseWidthSet(ADC_PWM_BASE, ADC_PWM_OUT, 3); //works freq = 20MHz
285  #endif
286 
287  // Start the timer.
288  MAP_PWMGenEnable(ADC_PWM_BASE, ADC_PWM_GEN);
289  // Enable the output.
290  MAP_PWMOutputState(ADC_PWM_BASE, ADC_PWM_OUT_BIT, true);
291  }

309 void ADC_initUDMAssiRX(void) {
310  SSIDMAEnable(ADC_SSI_BASE, SSI_DMA_RX);
311 
312  uDMAChannelAssign(ADC_SSI_RX_UDMA_CHANNEL_ASSIGN);
313 
314  // Place the uDMA channel attributes in a known state. These should already be disabled by default.
315  uDMAChannelAttributeDisable(ADC_SSI_RX_UDMA_CHANNEL,
316  UDMA_ATTR_USEBURST | UDMA_ATTR_ALTSELECT |
317  (UDMA_ATTR_HIGH_PRIORITY |
318  UDMA_ATTR_REQMASK));
319  // Configure the control parameters for the SSI2 RX channel. The channel
320  // will be used to transfer the ADC measurements to memory.
321  uDMAChannelControlSet(ADC_SSI_RX_UDMA_CHANNEL | UDMA_PRI_SELECT,
322  UDMA_SIZE_8 | UDMA_SRC_INC_NONE | UDMA_DST_INC_8 |
323  UDMA_ARB_4);
324  // Set up the transfer parameters for the SSI2 Rx channel. This will
325  // configure the transfer buffers and the transfer size.
326  uDMAChannelTransferSet(ADC_SSI_RX_UDMA_CHANNEL | UDMA_PRI_SELECT,
327  UDMA_MODE_BASIC,
328  (void *)(ADC_SSI_BASE + SSI_O_DR), (void *)(ADC_g_dataBufferBytes),
329  3);
330  uDMAChannelAttributeEnable(ADC_SSI_RX_UDMA_CHANNEL, UDMA_ATTR_HIGH_PRIORITY);
331  SSIIntEnable(ADC_SSI_BASE, SSI_DMARX);
332  IntEnable(ADC_SSI_INT);
333 
334  uDMAChannelEnable(ADC_SSI_RX_UDMA_CHANNEL);
335 }
