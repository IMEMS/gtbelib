119 int32_t twe_initFlash(uint32_t codeStart, uint32_t codeReserveLength,
120  uint32_t dataStart, uint32_t dataEraseLength) {
121  int32_t status = 0;
122  //protect Code area
123  status = twe_protectFlashRange(codeStart, codeReserveLength, FlashExecuteOnly);
124  // Erase Data in flash memory
125  status = twe_eraseFlashRange(dataStart, dataEraseLength);
126  //setup Flash Interrupt to handle any improper write attempts to flash
127  FlashIntEnable(FLASH_INT_PROGRAM);
128  return status;

139 int32_t twe_eraseFlashRange(uint32_t startAddress, uint32_t eraseLength) {
140  int32_t status = 0;
141  uint32_t modCheck;
142  uint32_t block;
143  modCheck = startAddress - startAddress/0x400;
144  modCheck = modCheck + eraseLength/0x400;
145  ASSERT(modCheck == 0);
146  if(eraseLength > 0x0) {
147  for(block = 0; block < (eraseLength/0x400); block++) {
148  status = FlashErase(startAddress + 0x400 * block) // Erases each 1KB block
149  ASSERT(status == 0);
150  }
151  }
152  return status;
153 }

163 int32_t twe_protectFlashRange(uint32_t startAddress, uint32_t protectLength, tFlashProtection eProtect) {
164  int32_t status = 0;
165  uint32_t modCheck;
166  uint32_t block;
167  // Check codeStart and codeReserveLength to ensure the protected memory
168  // block starts and ends on a 2KB boundry
169  modCheck = startAddress - startAddress/0x800;
170  modCheck = modCheck + protectLength/0x800;
171  ASSERT(modCheck == 0);
172  // Setup code protection
173  if(protectLength > 0x0) {
174  for(block = 0; block < (protectLength/0x800); block++) {
175  status = FlashProtectSet(0x800 * block, eProtect);
176  ASSERT(status == 0);
177  }
178  }
179  return status;
180 }
