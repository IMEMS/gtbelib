2  *
3  * adc_ads1278.h - headers for the TI ads1278 ADC.
4  * Interfaces the TI tiva C Launchpad with the TI ads1278 analog to digital
5  * converter
6  *
7  * Author: Curtis Mayberry
8  * Georgia Tech IMEMS
9  * February 2014
10  *
11  * Originally written for the MRIG gyroscope project
12  *
13  * GT BE Peripherals:
14  * ADC: ADS1278
15  * INPUTS
16  * TEST0 -> GPIO PE4: PE4
17  * TEST1 -> GPIO PE4: PE4
18  * CLKDIV -> GPIO PE5: PE5
19  * ~SYNC -> GPIO PE0: PE0
20  * CLK -> M1PWM5 : PF1 (option: hardware jumper)
21  * CLK -> SSI2_CLK: PB4 (option: hardware jumper)
22  * MODE0 -> GPIO PA6: PA6
23  * MODE1 -> GPIO PA7: PA7
24  * FORMAT0 -> GPIO PC4: PC4
25  * FORMAT1 -> GPIO PC5: PC5
26  * FORMAT2 -> GPIO PC6: PC6
27  * SERIAL
28  * SCLK -> SSI2_CLK: PB4
29  * ~DRDY/ FSYNC -> GPIO PB5: PB5 (CS) (SPI Format: ~DRDY)
30  * DOUT1 -> SSI2_RX: PB6 (MISO)
31  * DOUT2 -> GPIO PB0: PB2
32  * DOUT3 -> GPIO PB1: PB1
33  * DOUT4 -> GPIO PB2: PB2
34  * DOUT5 -> GPIO PB3: PB3
35  * DOUT6 -> GPIO PB7: PB7
36  * DOUT7 -> GPIO PD6: PD6
37  * DOUT8 -> GPIO PD7: PD7
38  *
39  * OUTPUTS
40  * NONE
41  *
