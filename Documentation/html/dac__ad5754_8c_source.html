412  void DAC_updateDataVolt(uint32_t dacAddress, uint32_t rangeValue, _Bool bin,
413  float voltage) {
414  uint32_t data;
415  if(dacAddress == DAC_ADDR_NONE_AD) {
416  DAC_writeNop();
417  }
418  else {
419  // Unipolar: Vout = Vref * Gain * D/(2^N)
420  // Bipolar : Vout = Vref * Gain * D/(2^N) - gain * Vref/2
421  switch(rangeValue) {
422  case DAC_RANGE_P5V : data = (uint32_t) (131072 * voltage + 5)/10; break;
423  case DAC_RANGE_P10V : data = (uint32_t) (65536 * voltage + 5)/10; break;
424  case DAC_RANGE_P10P8V : data = (uint32_t) (327680 * voltage + 27)/54; break;
425  case DAC_RANGE_PM5V : data = (uint32_t) (65536 * (voltage + 5) + 5)/10; break;
426  case DAC_RANGE_PM10V : data = (uint32_t) (32768 * (voltage + 10) + 5)/10; break;
427  case DAC_RANGE_PM10P8V : data = (uint32_t) (81920 * voltage + 884763)/432; break;
428  default : data = 0;
429  }
430  // Convert to two's complement from binary offset format
431  if(!bin) {
432  data ^= 0x80000000; //inverts MSB (sign bit)
433  }
434 
435  MAP_SSIDataPut(DAC_SSI_BASE, dacAddress); // Input reg Command
436  MAP_SSIDataPut(DAC_SSI_BASE, (data >> 8)); // First data byte
437  MAP_SSIDataPut(DAC_SSI_BASE, data); // Second data byte
438  }
439  }

528 void DAC_inituDMAautoSSI(void) {
529  // Enable the uDMA controller error interrupt. This interrupt will occur
530  // if there is a bus error during a transfer.
531  //MAP_IntEnable(INT_UDMAERR);
532 
533  // Enable the uDMA controller.
534  //uDMAEnable();
535 
536  //Setup channel selection
537  uDMAChannelAssign(DAC_SSI_TX_UDMA_CHANNEL_SELECTION);
538 
539  // Place the uDMA channel attributes in a known state. These should already be disabled by default.
540  uDMAChannelAttributeDisable(DAC_SSI_TX_UDMA_CHANNEL ,
541  UDMA_ATTR_USEBURST | UDMA_ATTR_ALTSELECT |
542  (UDMA_ATTR_HIGH_PRIORITY |
543  UDMA_ATTR_REQMASK));
544  // Configure the control parameters for the SSI0 TX channel. The channel
545  // will be used to update the dac output
546  uDMAChannelControlSet(DAC_SSI_TX_UDMA_CHANNEL | UDMA_PRI_SELECT,
547  UDMA_SIZE_8 | UDMA_SRC_INC_8 | UDMA_DST_INC_NONE |
548  UDMA_ARB_8);
549 
550  // Set up the transfer parameters for the SSI0 Tx channel. This will
551  // configure the transfer buffers and the transfer size.
552  uDMAChannelTransferSet(DAC_SSI_TX_UDMA_CHANNEL | UDMA_PRI_SELECT,
553  UDMA_MODE_AUTO,
554  (void *)DAC_g_bufferPRI, (void *)(DAC_SSI_BASE + SSI_O_DR),
