87 void temp_configSensorContinuousRead(uint32_t *statusreg) {
88  uint32_t commandWriteConfigReg = 0x00000C01;
89  uint32_t commandReadStatusReg = 0x00000040; // 0b01000000
90  uint32_t commandReadStatusEmpty = 0x00000000;
91  uint32_t trashBin[2];
92 
93  // Setup Configuration Register (register address: 0x01
94  SSIDataPut(SSI0_BASE, commandWriteConfigReg);
95  SSIDataPut(SSI0_BASE, commandReadStatusEmpty);
96  // SysCtlDelay(4170000); // (3 cycles/loop) * (1/50MHz) * (4170000 loop cycles) = 250.2ms
97 
98  // Read any residual data from the SSI port. This makes sure the receive
99  // FIFOs are empty, so we don't read any unwanted junk. This is done here
100  // because the SPI SSI mode is full-duplex, which allows you to send and
101  // receive at the same time. The SSIDataGetNonBlocking function returns
102  // "true" when data was returned, and "false" when no data was returned.
103  // The "non-blocking" function checks if there is any data in the receive
104  // FIFO and does not "hang" if there isn't.
105  while(SSIDataGetNonBlocking(SSI0_BASE, &trashBin[0]))
106  {
107  }
108 
109  // Read status register (Address: 0x0)
110  SSIDataPut(SSI0_BASE, commandReadStatusReg);
111  SSIDataGet(SSI0_BASE, &trashBin[1]);
112  SSIDataPut(SSI0_BASE, commandReadStatusEmpty);
113  SSIDataGet(SSI0_BASE, statusreg);
114 
115  //Need to adjust for different processor frequencies
116  SysCtlDelay(4170000); // (3 cycles/loop) * (1/50MHz) * (4170000 loop cycles) = 250.2ms
117 }

123 void temp_resetSPI(void) {
124  uint32_t commandReset = 0x0000FFFF;
125  uint32_t trashBin[1];
126 
127  // Reset by 32 cycles of 1 on Din
128  SSIDataPut(SSI0_BASE, commandReset);
129  SSIDataPut(SSI0_BASE, commandReset);
130 
131  // Empty Rx FIFO
132  while(SSIDataGetNonBlocking(SSI0_BASE, &trashBin[0])) {
133  }
134 
135  // Delay for the required 500us
136  SysCtlDelay(8334); // (3 cycles/loop) * (1/50MHz) * (4170000 loop cycles) = 600us (500us needed after reset)
137 }

145 float temp_degC13bitReading(uint32_t reading) {

163 float temp_degF13bitReading(uint32_t reading) {
164  reading = reading>>3;
165  if (reading < 8192) {
166  // Positive Reading
167  return (9*reading+2560)/80; // degF = (9/5)*degC + 32
168  }
169  else {
170  // Negative Reading
171  return (9*reading-5632)/80;
172  }
173 }

208 void temp_initSSI0uDMA(void) {
209  uint32_t trashBin[1];
210  // Enable Peripherals
211  ROM_SysCtlPeripheralEnable(SYSCTL_PERIPH_SSI0);
212  ROM_SysCtlPeripheralEnable(SYSCTL_PERIPH_GPIOA);
213 
214  // Set the pin muxing to SSI0 on port A
215  GPIOPinConfigure(GPIO_PA2_SSI0CLK);
216  GPIOPinConfigure(GPIO_PA3_SSI0FSS);
217  GPIOPinConfigure(GPIO_PA4_SSI0RX);
218  GPIOPinConfigure(GPIO_PA5_SSI0TX);
219  GPIOPinTypeSSI(GPIO_PORTA_BASE,GPIO_PIN_5|GPIO_PIN_4|GPIO_PIN_3|GPIO_PIN_2);
220 
221  // Bit rate:10,000 baud
222  // SPI Mode3
223  SSIConfigSetExpClk(SSI0_BASE,SysCtlClockGet(),SSI_FRF_MOTO_MODE_3,SSI_MODE_MASTER,10000,16);
224 
225  // Enable SSI DMA
226  SSIEnable(SSI0_BASE);
227  SSIDMAEnable(SSI0_BASE, SSI_DMA_RX);
228  while (ROM_SSIDataGetNonBlocking(SSI0_BASE, trashBin)) {
229  }
230  ROM_IntEnable(INT_SSI0);
231 }

241 void temp_initTimer2A(void) {
242  uint32_t timerCount = 0;
243 
244  SysCtlPeripheralEnable(SYSCTL_PERIPH_TIMER2);
245  TimerConfigure(TIMER2_BASE, TIMER_CFG_PERIODIC); // Full-width periodic timer
246 
247  // Set the timer period to 250ms
248  timerCount = (SysCtlClockGet()) /12500000;
249  TimerLoadSet(TIMER2_BASE, TIMER_A, timerCount -1);
250 
251  //Start Timer Interrupt
252  IntEnable(INT_TIMER2A);
253  TimerIntEnable(TIMER2_BASE, TIMER_TIMA_TIMEOUT);
254  IntMasterEnable();
255 
256  TimerEnable(TIMER2_BASE, TIMER_A);
257 }
